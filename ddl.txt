CREATE TABLE IF NOT EXISTS survey (
    survey_id VARCHAR(50) PRIMARY KEY,
    title VARCHAR(200) NOT NULL,
    service VARCHAR(100) NOT NULL,
    survey_type VARCHAR(50) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS survey_question (
    survey_id VARCHAR(50) NOT NULL,
    question_code VARCHAR(20) NOT NULL,
    question_type VARCHAR(20) NOT NULL,
    category VARCHAR(100) NOT NULL,
    question_text TEXT NOT NULL,
    display_order INTEGER NOT NULL,
    PRIMARY KEY (survey_id, question_code)
);

CREATE TABLE IF NOT EXISTS survey_response (
    response_id VARCHAR(50) PRIMARY KEY,
    survey_id VARCHAR(50) NOT NULL,
    submitted_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS survey_answer (
    answer_id VARCHAR(50) PRIMARY KEY,
    response_id VARCHAR(50) NOT NULL,
    survey_id VARCHAR(50) NOT NULL,
    question_code VARCHAR(20) NOT NULL,
    category VARCHAR(100) NOT NULL,
    answer_type VARCHAR(20) NOT NULL,
    score_value INTEGER,
    text_value TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS text_analysis (
    answer_id VARCHAR(50) PRIMARY KEY,
    survey_id VARCHAR(50) NOT NULL,
    question_code VARCHAR(20) NOT NULL,
    category VARCHAR(100) NOT NULL,
    sentiment VARCHAR(20),
    pos_score NUMERIC(10, 6),
    neu_score NUMERIC(10, 6),
    neg_score NUMERIC(10, 6),
    mix_score NUMERIC(10, 6),
    keywords JSONB,
    analyzed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS survey_report (
    report_id VARCHAR(50) PRIMARY KEY,
    survey_id VARCHAR(50) NOT NULL,
    report_json JSONB NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
